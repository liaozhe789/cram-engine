入口参数：学号
1 显示学生的选课信息
2 如果学生没选课，让他选1号课，插入一条选课记录，成绩为0
3 如果学生所有课都不及格，删掉该学生记录

代码：
CREATE PROCEDURE usp_ProcessStudent
    @sid char(10)
AS
BEGIN
    DECLARE @total_courses int;
    DECLARE @passed_courses int;

    SELECT Student.学号, Student.姓名, Course.课名, SC.成绩
    FROM Student
    JOIN SC ON Student.学号 = SC.学号
    JOIN Course ON SC.课号 = Course.课号
    WHERE Student.学号 = @sid;

    SELECT @total_courses = COUNT(*) FROM SC WHERE 学号 = @sid;

    IF @total_courses = 0
    BEGIN
        INSERT INTO SC (学号, 课号, 成绩) 
        VALUES (@sid, 'C01', 0);
    END
    ELSE
    BEGIN
        SELECT @passed_courses = COUNT(*) FROM SC WHERE 学号 = @sid AND 成绩 >= 60;

        IF @passed_courses = 0
        BEGIN
            DELETE FROM SC WHERE 学号 = @sid;
            DELETE FROM Student WHERE 学号 = @sid;
        END
    END
END;
