-- 1. 查询选修了“数据库原理”的学生的人数和平均成绩
SELECT COUNT(*) AS 学生人数, AVG(成绩) AS 平均成绩
FROM SC, Course
WHERE SC.课号 = Course.课号 AND Course.课名 = N'数据库原理';


-- 2. 查询选修 3 门课以上的学生姓名
SELECT Student.姓名
FROM Student, SC
WHERE Student.学号 = SC.学号
GROUP BY Student.姓名, Student.学号
HAVING COUNT(SC.课号) > 3;


-- 3. 列出人数超过 1000 人的系名
SELECT 系号, COUNT(*) AS 总人数
FROM Student
GROUP BY 系号
HAVING COUNT(*) > 2; 


-- 4. 查询所有课程都及格的学生姓名
SELECT 姓名
FROM Student
