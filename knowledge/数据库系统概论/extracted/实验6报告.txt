实验内容与过程
学生（学号，年龄，性别，系名） 
课程（课号，课名，学分，学时）
选课（学号，课号，成绩）
1．建立学生表的insert触发器，若向学生表中插入一条记录，则自动向选课表中插入一行，令该生选一门选修人数最多的课。
2．在建立学生表的delete触发器，若选课表中该生有选课记录，则拒绝删除，要求在删除学生信息同时，将相关表中的信息全部删除。
代码：
CREATE TRIGGER tri_Student_Insert
ON Student
AFTER INSERT
AS
BEGIN
    -- 定义变量存放新同学的学号和人数最多的课号
    DECLARE @new_sid char(10);
    DECLARE @max_cid char(10);

    -- 从 Inserted 临时表中获取刚插入的学生学号
    SELECT @new_sid = 学号 FROM Inserted;

    -- 通过子查询找出选课人数最多（出现次数最多）的课号
    SELECT TOP 1 @max_cid = 课号 
    FROM SC 
    GROUP BY 课号 
    ORDER BY COUNT(学号) DESC;

    -- 自动向选课表插入关联行
    IF @max_cid IS NOT NULL
    BEGIN
        INSERT INTO SC (学号, 课号, 成绩) 
        VALUES (@new_sid, @max_cid, NULL);
    END
END;

INSERT INTO Student (学号, 年龄, 性别, 系号, 姓名)

-- 检查选课表，看系统是否自动帮他选了“数据库原理(C01)”
SELECT * FROM SC WHERE 学号 = '2026006';

题目2：
CREATE TRIGGER tri_Student_Delete
ON Student
INSTEAD OF DELETE
AS
BEGIN
    --声明变量存放即将被删除的学号
    DECLARE @del_sid char(10);
    SELECT @del_sid = 学号 FROM Deleted;

    --级联删除：先去彻底清除该学生在 SC 表里的历史选课记录
    DELETE FROM SC WHERE 学号 = @del_sid;

    -- 清理干净后，安全地将该学生从 Student 表中抹去
    DELETE FROM Student WHERE 学号 = @del_sid;
    END;

实验心得
本次实验聚焦于触发器的设计与应用，使我深入掌握了使用 Inserted 和 Deleted 临时表处理动态数据的核心技术。在编写 INSERT 触发器时，我学会了如何通过逻辑派生，结合聚合函数实现复杂的自动化业务托管，让系统自适应为新学生补全选课记录。
